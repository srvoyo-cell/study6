Листинг 1.2. Пример очистки данных
```
# Более комплексный пример очистки
dirty_text = """
<p>Статья о Python. Посетите сайт https://python.org.
Цена товара 1,234.50 руб. Пишите на email@example.com.
"""
def clean_text(text):
 # 1. Приводим к нижнему регистру
 text = text.lower()
 # 2. Удаляем HTML-теги
 text = re.sub(r'<[^>]+>', ' ', text)
 # 3. Заменяем URLs, emails, числа на специальные метки или удаляем
 text = re.sub(r'https?://\S+|www\.\S+', ' URL ', text)
 text = re.sub(r'\S+@\S+', ' EMAIL ', text)
 text = re.sub(r'\d+[.,]?\d*', ' NUM ', text)
 # 4. Удаляем всю пунктуацию, кроме дефиса (для составных слов)
 text = re.sub(r'[^\w\s-]', ' ', text)
 # 5. Заменяем переносы строк и табы на пробелы
 text = re.sub(r'\s+', ' ', text)
 # 6. Убираем пробелы в начале и конце
 text = text.strip()
 return text
cleaned = clean_text(dirty_text)
print(cleaned)
```

Более сложный статистический анализ текста реализован в библиотеках с collections,
NLTK, spaCy.
Библиотека collections содержит класс Counter для подсчета частот.
Библиотеки NLTK (nltk.download('punkt'), nltk.download('stopwords')) или
spaCy являются более тяжеловесными альтернативами. В них реализованы подсчет базовой
статистики: количество документов, средняя длина, распределение частот слов.

Листинг 1.4. Обработка текста с использоваием библиотек
```
from collections import Counter
import nltk
from nltk.tokenize import word_tokenize
# nltk.download('punkt') # Если не скачано
# nltk.download('stopwords') # Скачиваем стоп-слова
from nltk.corpus import stopwords
# Объединим все очищенные тексты в один большой список слов
all_words = []
for text in df['text_clean']:
 # Используем NLTK для токенизации (учитывает язык)
 tokens = word_tokenize(text, language='russian') # или 'english'
 all_words.extend(tokens)
print(f"Всего слов (токенов): {len(all_words)}")
print(f"Уникальных слов: {len(set(all_words))}")
# 1. Подсчет частот
word_freq = Counter(all_words)
print("10 самых частых слов:", word_freq.most_common(10))
# 2. Анализ длины документов
df['doc_length'] = df['text_clean'].apply(lambda x: len(word_tokenize(x, langua
ge='russian')))
print(f"\nСредняя длина документа: {df['doc_length'].mean():.1f} слов")
print(f"Медианная длина документа: {df['doc_length'].median():.1f} слов")
# 3. Частотное распределение без стоп-слов
stop_words = set(stopwords.words('russian'))
filtered_words = [word for word in all_words if word not in stop_words and len(
word) > 2]
filtered_freq = Counter(filtered_words)
print("\n10 самых частых слов без стоп-слов:", filtered_freq.most_common(10))
```
